module sclkGen(
    input trig, 
    input clk, 
    input rst,
    output sclk 
);

  reg [1:0] state; 
  reg [3:0] counter;
  reg sclk_reg; 
  
  localparam IDLE = 0; 
  localparam OUTPUT = 1; 
  localparam STOP = 2; 

  always @(posedge clk) begin

    if (rst)
        state <= IDLE;

    case (state)
    
      IDLE: begin
        sclk_reg <= 0; 
        counter <= 0;
        if (trig)
          state <= OUTPUT; 
      end
      
      OUTPUT: begin
        sclk_reg = ~sclk_reg;
        counter <= counter + 1; 
        if (counter == 8) begin
          counter <= 0; 
          sclk_reg <= 0;
          state <= STOP;
        end
      end
      
      STOP: if (trig==0)                
                state <= IDLE;
                
      default: state <= IDLE;
      
    endcase
  end  
  
  assign sclk = sclk_reg; 
  
endmodule

module sclkGen_tb;
  
  reg trig;
  reg clk;
  reg rst;
  wire sclk;
  
  sclkGen dut (
    .trig(trig),
    .clk(clk),
    .rst(rst),
    .sclk(sclk)
  );
  
  always #50 clk = ~clk;
  
  initial begin   
    trig = 0;
    clk = 0;
    rst = 0;
    #50;
    rst = 1;
    #60;
    rst = 0;
    
    repeat(3) begin
        trig=1;
        #({$random}%500+1000);
        trig=0;
        #({$random}%500+200);
    end   
    #1000;
    
    $finish;
  end
  
endmodule
